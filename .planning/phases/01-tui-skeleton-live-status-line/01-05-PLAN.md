          try:

              console.print(f"[dim]TUI unavailable ({type(exc).__name__}); using plain REPL.[/dim]")

              try:
                  run_tui(cfg, resume=resume)
                  return
              except Exception as exc:  # terminal can't host Textual, etc. -> fall back
                  console.print(

                  )
      # fall through to the plain loop
